\section{Expected value}
    \subsection*{Def}
        $E[X]=\sum x_i*p(x_i)$

    \subsection*{E.V by CDF}
        \[E[X]=\sum_{i=1}^{\infty} (x_i-x_{i-1})*(1-F_X(x_{i-1}))\]

    \subsection*{LOTUS}
        $E[g(X)]=\sum g(x_i)*p(x_i)$

    \subsection*{Linear}
        $E[g(x) + h(x)] = E[g(x)] + E[h(x)]$
