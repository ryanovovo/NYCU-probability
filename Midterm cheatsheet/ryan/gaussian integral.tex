\section{Gaussian Integral}
    \begin{enumerate}
        \item $\int_{-\infty}^{\infty} e^{-ax^2} dx = \sqrt{\frac{\pi}{a}}$
        \item $\int_{-\infty}^{\infty} x e^{-a(x-b)^2} dx = b \sqrt{\frac{\pi}{a}}$
        \item $\int_{-\infty}^{\infty} x^2 e^{-ax^2} dx = \frac{\sqrt{\pi}}{2a^{3/2}}$
    \end{enumerate}
