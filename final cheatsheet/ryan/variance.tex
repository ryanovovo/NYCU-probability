\section{Variance}
    \subsection*{Def}
        $Var(X)=E[(X-E[X])^2]=E[X^2]-E[X]^2$

    \subsection*{Properties}
        \begin{enumerate}
            \item $Var(X) \geq 0$
            \item $Var(X) = 0 \iff P(X=E[X]) = 1$
            \item $Var(X+c) = Var(X)$
            \item $Var(aX)=a^2Var(X)$
            \item $Var(X+Y) = Var(X) + Var(Y)$
            
            (X, Y are independent)
        \end{enumerate}
